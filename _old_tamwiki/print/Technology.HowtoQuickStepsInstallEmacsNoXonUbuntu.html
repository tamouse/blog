Obtaining and installing emacs24
  1. # Obtain latest emacs from Savannah: http://ftp.gnu.org/gnu/emacs/
  2. wget http://ftp.gnu.org/gnu/emacs/emacs-24.3.tar.gz
  3.  
  4. # Install dependencies:
  5. sudo apt-get install libjpeg-dev libpng-dev libgif-dev libtiff-dev libncurses-dev -y
  6.  
  7. # Untar the archive:
  8. tar xvfz emacs-24.3-rc.tar.gz
  9.  
  10. cd emacs-24.3
  11.  
  12. # configure without x
  13. ./configure --without-x
  14.  
  15. make
  16.  
  17. sudo make install
